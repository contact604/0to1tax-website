## ⚙️ JavaScript 로직

### 탭 전환 함수

```javascript
window.switchGrowthTab = function(tab) {
    const basicTab = document.getElementById('growthBasicTab');
    const additionalTab = document.getElementById('growthAdditionalTab');
    const basicBtn = document.getElementById('growthTabBasic');
    const additionalBtn = document.getElementById('growthTabAdditional');

    if (tab === 'basic') {
        // 기본 서비스 탭 활성화
        basicTab.classList.remove('hidden');
        additionalTab.classList.add('hidden');
        
        // 버튼 스타일 변경
        basicBtn.classList.add('bg-white', 'text-stellar-purple');
        basicBtn.classList.remove('bg-opacity-20', 'text-white');
        
        additionalBtn.classList.add('bg-opacity-20', 'text-white');
        additionalBtn.classList.remove('text-stellar-purple');
    } else {
        // 추가 서비스 탭 활성화
        // (역순 동작)
    }
}
```
